    <script>
        // ข้อมูลสำหรับเกม
        const vowelSounds = {
            a: ["cat", "hat", "bat", "map", "apple"],
            e: ["bed", "red", "pen", "leg", "elephant"],
            i: ["sit", "pig", "fish", "lip", "pin"],
            o: ["dog", "log", "box", "top", "pot"],
            u: ["sun", "bus", "cup", "duck", "umbrella"]
        };

        // ฟังก์ชันสำหรับเล่นเสียง
        function playSound(text, rate = 0.8, pitch = 1.0) {
            const utterance = new SpeechSynthesisUtterance(text);
            utterance.lang = 'en-US';
            utterance.rate = rate;
            utterance.pitch = pitch;
            window.speechSynthesis.speak(utterance);
        }

        // ฟังก์ชันย้อนกลับ
        function goBack() {
            alert('กำลังกลับไปหน้า Dashboard...');
            // ในเว็บจริง: window.location.href = 'dashboard.html';
        }

        // ตั้งค่า event listeners เมื่อโหลดหน้าเสร็จ
        document.addEventListener('DOMContentLoaded', function() {
            // ตั้งค่า event สำหรับปุ่มเสียง
            document.querySelectorAll('.sound-button').forEach(button => {
                button.addEventListener('click', function(event) {
                    event.stopPropagation();
                    
                    const card = this.closest('.vowel-header');
                    if (card) {
                        const vowel = card.querySelector('.vowel-icon').textContent;
                        
                        if (this.classList.contains('vowel')) {
                            // เล่นเสียงสระ
                            playSound(vowel, 0.7, 1.2);
                        } else {
                            // เล่นคำอธิบาย
                            const description = card.querySelector('.vowel-description').textContent;
                            playSound(description, 0.6, 1.0);
                        }
                        
                        // เอฟเฟ็กต์ปุ่ม
                        this.classList.add('playing');
                        setTimeout(() => {
                            this.classList.remove('playing');
                        }, 1000);
                    }
                });
            });
            
            // ตั้งค่า event สำหรับปุ่มฟังเสียงคำศัพท์
            document.querySelectorAll('.word-sound').forEach(button => {
                button.addEventListener('click', function() {
                    const word = this.getAttribute('data-word');
                    playSound(word, 0.7, 1.0);
                    
                    // เอฟเฟ็กต์ปุ่ม
                    this.classList.add('playing');
                    setTimeout(() => {
                        this.classList.remove('playing');
                    }, 1000);
                });
            });
            
            // เกมฟังเสียงเลือกสระ
            let currentSound = null;
            const soundResult = document.getElementById('sound-result');
            
            // สุ่มคำศัพท์สำหรับเกม
            function getRandomWord() {
                const vowels = Object.keys(vowelSounds);
                const randomVowel = vowels[Math.floor(Math.random() * vowels.length)];
                const words = vowelSounds[randomVowel];
                const randomWord = words[Math.floor(Math.random() * words.length)];
                
                return {
                    vowel: randomVowel,
                    word: randomWord
                };
            }
            
            // เล่นเสียงสำหรับเกม
            document.getElementById('play-game-sound').addEventListener('click', function() {
                currentSound = getRandomWord();
                playSound(currentSound.word, 0.7, 1.0);
                soundResult.textContent = '';
                
                // รีเซ็ตการเลือก
                document.querySelectorAll('.vowel-option').forEach(option => {
                    option.classList.remove('selected');
                });
            });
            
            // เลือกสระ
            document.querySelectorAll('.vowel-option').forEach(option => {
                option.addEventListener('click', function() {
                    // รีเซ็ตการเลือก
                    document.querySelectorAll('.vowel-option').forEach(opt => {
                        opt.classList.remove('selected');
                    });
                    
                    // เลือกอันนี้
                    this.classList.add('selected');
                });
            });
            
            // ตรวจคำตอบ
            document.getElementById('check-answer').addEventListener('click', function() {
                if (!currentSound) {
                    soundResult.textContent = 'กรุณากดฟังเสียงก่อนตรวจสอบ!';
                    soundResult.style.color = '#ff6b6b';
                    return;
                }
                
                const selectedOption = document.querySelector('.vowel-option.selected');
                if (!selectedOption) {
                    soundResult.textContent = 'กรุณาเลือกสระก่อนตรวจสอบ!';
                    soundResult.style.color = '#ff6b6b';
                    return;
                }
                
                const selectedVowel = selectedOption.getAttribute('data-vowel');
                if (selectedVowel === currentSound.vowel) {
                    soundResult.textContent = 'ถูกต้อง! คุณเก่งมาก!';
                    soundResult.style.color = '#4a69bd';
                } else {
                    soundResult.textContent = `ยังไม่ถูกต้อง! คำนี้มีสระเสียงสั้น '${currentSound.vowel}'`;
                    soundResult.style.color = '#ff6b6b';
                }
            });
            
            // เกมเติมสระในคำ
            const puzzleContainer = document.getElementById('puzzle-container');
            const puzzleResult = document.getElementById('puzzle-result');
            let currentPuzzle = null;
            
            // สร้างปริศนา
            function createPuzzle() {
                const puzzle = getRandomWord();
                currentPuzzle = puzzle;
                
                // แยกคำออกเป็นตัวอักษร
                const letters = puzzle.word.split('');
                const vowelIndex = letters.findIndex(letter => puzzle.vowel === letter);
                
                // สร้าง HTML สำหรับปริศนา
                let puzzleHTML = `
                    <div class="puzzle-image">${getEmojiForWord(puzzle.word)}</div>
                    <div class="word-puzzle">
                `;
                
                letters.forEach((letter, index) => {
                    if (index === vowelIndex) {
                        puzzleHTML += `<div class="puzzle-blank">?</div>`;
                    } else {
                        puzzleHTML += `<div class="puzzle-letter">${letter}</div>`;
                    }
                });
                
                puzzleHTML += `</div>`;
                puzzleContainer.innerHTML = puzzleHTML;
                puzzleResult.textContent = '';
                
                // รีเซ็ตการเลือก
                document.querySelectorAll('.vowel-option').forEach(option => {
                    option.classList.remove('selected');
                });
            }
            
            // ฟังก์ชันช่วยเหลือ: หา emoji จากคำ
            function getEmojiForWord(word) {
                const emojiMap = {
                    'cat': '🐱',
                    'hat': '🎩',
                    'bat': '🦇',
                    'map': '🗺️',
                    'apple': '🍎',
                    'bed': '🛏️',
                    'red': '🔴',
                    'pen': '✏️',
                    'leg': '🦵',
                    'elephant': '🐘',
                    'sit': '💺',
                    'pig': '🐷',
                    'fish': '🐟',
                    'lip': '👄',
                    'pin': '📌',
                    'dog': '🐶',
                    'log': '🪵',
                    'box': '📦',
                    'top': '🔄',
                    'pot': '🍲',
                    'sun': '☀️',
                    'bus': '🚌',
                    'cup': '☕',
                    'duck': '🦆',
                    'umbrella': '☂️'
                };
                
                return emojiMap[word] || '❓';
            }
            
            // เลือกสระสำหรับเกมเติมคำ
            document.querySelectorAll('#vowel-options .vowel-option').forEach(option => {
                option.addEventListener('click', function() {
                    // รีเซ็ตการเลือก
                    document.querySelectorAll('#vowel-options .vowel-option').forEach(opt => {
                        opt.classList.remove('selected');
                    });
                    
                    // เลือกอันนี้
                    this.classList.add('selected');
                });
            });
            
            // ตรวจสอบคำตอบเกมเติมคำ
            document.getElementById('next-puzzle').addEventListener('click', function() {
                if (!currentPuzzle) {
                    createPuzzle();
                    return;
                }
                
                const selectedOption = document.querySelector('#vowel-options .vowel-option.selected');
                if (!selectedOption) {
                    puzzleResult.textContent = 'กรุณาเลือกสระก่อนตรวจสอบ!';
                    puzzleResult.style.color = '#ff6b6b';
                    return;
                }
                
                const selectedVowel = selectedOption.getAttribute('data-vowel');
                if (selectedVowel === currentPuzzle.vowel) {
                    puzzleResult.textContent = 'ถูกต้อง! เยี่ยมมาก!';
                    puzzleResult.style.color = '#4a69bd';
                    setTimeout(createPuzzle, 1500);
                } else {
                    puzzleResult.textContent = `ลองใหม่อีกครั้ง! คำนี้คือ "${currentPuzzle.word}"`;
                    puzzleResult.style.color = '#ff6b6b';
                }
            });
            
            // สร้างปริศนาแรก
            createPuzzle();
        });
