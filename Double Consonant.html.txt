    <script>
        // ข้อมูลสำหรับเกม
        const doubleConsonantWords = [
            { word: "rabbit", consonant: "bb", image: "🐇" },
            { word: "bubble", consonant: "bb", image: "🫧" },
            { word: "ladder", consonant: "dd", image: "🪜" },
            { word: "paddle", consonant: "dd", image: "🛶" },
            { word: "coffee", consonant: "ff", image: "☕" },
            { word: "office", consonant: "ff", image: "🏢" },
            { word: "ball", consonant: "ll", image: "⚽" },
            { word: "hello", consonant: "ll", image: "👋" },
            { word: "class", consonant: "ss", image: "👨‍🏫" },
            { word: "dress", consonant: "ss", image: "👗" },
            { word: "pizza", consonant: "zz", image: "🍕" },
            { word: "jazz", consonant: "zz", image: "🎷" }
        ];

        // ฟังก์ชันสำหรับเล่นเสียง
        function playSound(text, rate = 0.8, pitch = 1.0) {
            const utterance = new SpeechSynthesisUtterance(text);
            utterance.lang = 'en-US';
            utterance.rate = rate;
            utterance.pitch = pitch;
            window.speechSynthesis.speak(utterance);
        }

        // ฟังก์ชันย้อนกลับ
        function goBack() {
            alert('กำลังกลับไปหน้า Dashboard...');
            // ในเว็บจริง: window.location.href = 'dashboard.html';
        }

        // ตั้งค่า event listeners เมื่อโหลดหน้าเสร็จ
        document.addEventListener('DOMContentLoaded', function() {
            // ตั้งค่า event สำหรับปุ่มเสียง
            document.querySelectorAll('.word-sound').forEach(button => {
                button.addEventListener('click', function() {
                    const word = this.getAttribute('data-word');
                    playSound(word, 0.7, 1.0);
                    
                    // เอฟเฟ็กต์ปุ่ม
                    this.style.backgroundColor = '#ff6b6b';
                    setTimeout(() => {
                        this.style.backgroundColor = '';
                    }, 500);
                });
            });
            
            // ตั้งค่า event สำหรับปุ่มเสียงพยัญชนะคู่
            document.querySelectorAll('.sound-button').forEach(button => {
                button.addEventListener('click', function() {
                    const card = this.closest('.consonant-card');
                    if (card) {
                        const consonant = card.querySelector('.consonant-title').textContent;
                        playSound(consonant, 0.6, 1.1);
                        
                        // เอฟเฟ็กต์ปุ่ม
                        this.style.backgroundColor = '#ff6b6b';
                        setTimeout(() => {
                            this.style.backgroundColor = '';
                        }, 500);
                    }
                });
            });
            
            // เกมเติมพยัญชนะคู่
            let currentPuzzle = null;
            const puzzleImage = document.getElementById('puzzle-image');
            const consonantBlank = document.getElementById('consonant-blank');
            const puzzleResult = document.getElementById('puzzle-result');
            let selectedConsonant = null;
            
            // สุ่มปริศนา
            function generatePuzzle() {
                const randomIndex = Math.floor(Math.random() * doubleConsonantWords.length);
                currentPuzzle = doubleConsonantWords[randomIndex];
                
                // แสดงภาพ
                puzzleImage.textContent = currentPuzzle.image;
                
                // รีเซ็ตการเลือก
                puzzleResult.textContent = '';
                selectedConsonant = null;
                document.querySelectorAll('.consonant-option').forEach(option => {
                    option.classList.remove('selected');
                });
            }
            
            // เลือกพยัญชนะ
            document.querySelectorAll('.consonant-option').forEach(option => {
                option.addEventListener('click', function() {
                    // รีเซ็ตการเลือก
                    document.querySelectorAll('.consonant-option').forEach(opt => {
                        opt.classList.remove('selected');
                    });
                    
                    // เลือกอันนี้
                    this.classList.add('selected');
                    selectedConsonant = this.getAttribute('data-consonant');
                    consonantBlank.textContent = selectedConsonant;
                });
            });
            
            // ตรวจสอบคำตอบ
            document.getElementById('check-puzzle').addEventListener('click', function() {
                if (!selectedConsonant) {
                    puzzleResult.textContent = 'กรุณาเลือกพยัญชนะคู่ก่อนตรวจสอบ!';
                    puzzleResult.style.color = '#ff6b6b';
                    return;
                }
                
                if (selectedConsonant === currentPuzzle.consonant) {
                    puzzleResult.textContent = 'ถูกต้อง! เยี่ยมมาก!';
                    puzzleResult.style.color = '#4a69bd';
                    playSound(currentPuzzle.word, 0.7, 1.0);
                } else {
                    puzzleResult.textContent = 'ยังไม่ถูกต้อง! ลองใหม่อีกครั้ง';
                    puzzleResult.style.color = '#ff6b6b';
                }
            });
            
            // คำถัดไป
            document.getElementById('next-puzzle').addEventListener('click', generatePuzzle);
            
            // เริ่มเกมแรก
            generatePuzzle();
            
            // เกมจับคู่พยัญชนะคู่
            const matchResult = document.getElementById('match-result');
            
            // ตั้งค่า drag and drop
            document.querySelectorAll('.word-option').forEach(option => {
                option.addEventListener('dragstart', function(e) {
                    e.dataTransfer.setData('text/plain', this.textContent);
                });
            });
            
            document.querySelectorAll('.consonant-group').forEach(group => {
                group.addEventListener('dragover', function(e) {
                    e.preventDefault();
                });
                
                group.addEventListener('drop', function(e) {
                    e.preventDefault();
                    const word = e.dataTransfer.getData('text/plain');
                    const correctConsonant = doubleConsonantWords.find(w => w.word === word)?.consonant;
                    const groupConsonant = this.getAttribute('data-consonant');
                    
                    if (correctConsonant === groupConsonant) {
                        this.style.backgroundColor = '#b5ead7';
                        matchResult.textContent = 'ถูกต้อง! ' + word + ' → ' + groupConsonant;
                        matchResult.style.color = '#4a69bd';
                        
                        // อ่านเสียงคำ
                        playSound(word, 0.7, 1.0);
                    } else {
                        this.style.backgroundColor = '#ffb7b2';
                        matchResult.textContent = 'ยังไม่ถูกต้อง! ลองใหม่อีกครั้ง';
                        matchResult.style.color = '#ff6b6b';
                    }
                    
                    setTimeout(() => {
                        this.style.backgroundColor = '';
                        matchResult.textContent = '';
                    }, 2000);
                });
            });
            
            // ปุ่มตรวจคำตอบเกมจับคู่
            document.getElementById('check-match').addEventListener('click', function() {
                matchResult.textContent = 'เกมนี้ต้องลากคำไปวางในกลุ่มพยัญชนะ!';
                matchResult.style.color = '#4a69bd';
            });
            
            // ปุ่มเริ่มใหม่
            document.getElementById('reset-match').addEventListener('click', function() {
                document.querySelectorAll('.consonant-group').forEach(group => {
                    group.style.backgroundColor = '';
                });
                matchResult.textContent = '';
            });
        });
    </script>
