    <script>
        // ฟังก์ชันสำหรับเล่นเสียงตัวอักษร
        function playLetterSound(letter) {
            // หาปุ่มที่กำลังเล่นอยู่และหยุดเล่น
            const currentPlaying = document.querySelector('.sound-button.playing');
            if (currentPlaying) {
                currentPlaying.classList.remove('playing');
            }
            
            // เริ่มเล่นเสียงใหม่
            const utterance = new SpeechSynthesisUtterance(letter);
            utterance.lang = 'en-US';
            utterance.rate = 0.8; // ความเร็วในการพูด
            utterance.pitch = 1.2; // ระดับเสียง
            
            // เพิ่มเอฟเฟ็กต์ให้ปุ่ม
            event.target.classList.add('playing');
            
            utterance.onend = function() {
                event.target.classList.remove('playing');
            };
            
            window.speechSynthesis.speak(utterance);
        }
        
        // ฟังก์ชันสำหรับเล่นเสียงคำศัพท์
        function playWordSound(word) {
            const utterance = new SpeechSynthesisUtterance(word);
            utterance.lang = 'en-US';
            utterance.rate = 0.7;
            window.speechSynthesis.speak(utterance);
        }
        
        // ฟังก์ชันสำหรับเล่นเสียง Phonics
        function playPhonicSound(phonicSymbol, phonicDesc) {
            const utterance = new SpeechSynthesisUtterance(phonicDesc);
            utterance.lang = 'en-US';
            utterance.rate = 0.6;
            utterance.pitch = 1.1;
            
            // เพิ่มเอฟเฟ็กต์ให้ปุ่ม
            event.target.classList.add('playing');
            
            utterance.onend = function() {
                event.target.classList.remove('playing');
            };
            
            window.speechSynthesis.speak(utterance);
        }
        
        // ฟังก์ชันย้อนกลับ
        function goBack() {
            alert('กำลังกลับไปหน้า Dashboard...');
            // ในเว็บจริง: window.location.href = 'dashboard.html';
        }
        
        // ฟังก์ชันตรวจสอบคำตอบ
        function checkWord() {
            const inputElement = document.getElementById('word-input');
            const word = inputElement.value.trim().toLowerCase();
            const targetLetter = document.getElementById('target-letter').textContent.toLowerCase();
            const resultElement = document.getElementById('result-message');
            
            if (!word) {
                resultElement.textContent = 'กรุณาพิมพ์คำศัพท์ก่อนตรวจสอบ!';
                resultElement.style.color = '#ff6b6b';
                return;
            }
            
            if (word.charAt(0) === targetLetter) {
                resultElement.textContent = 'ถูกต้อง! "' + word + '" ขึ้นต้นด้วยตัวอักษร ' + targetLetter.toUpperCase();
                resultElement.style.color = '#4a69bd';
                playWordSound(word);
                
                // สุ่มตัวอักษรใหม่
                const letters = 'abcdefghijklmnopqrstuvwxyz';
                const randomLetter = letters[Math.floor(Math.random() * letters.length)];
                document.getElementById('target-letter').textContent = randomLetter.toUpperCase();
            } else {
                resultElement.textContent = 'คำนี้ไม่ขึ้นต้นด้วย ' + targetLetter.toUpperCase() + '! ลองใหม่อีกครั้ง';
                resultElement.style.color = '#ff6b6b';
            }
            
            // ล้างช่อง input
            inputElement.value = '';
            inputElement.focus();
        }
        
        // ตั้งค่า event listeners เมื่อโหลดหน้าเสร็จ
        document.addEventListener('DOMContentLoaded', function() {
            // ตั้งค่า event สำหรับการ์ดตัวอักษร
            const letterCards = document.querySelectorAll('.letter-card');
            letterCards.forEach(card => {
                const letter = card.getAttribute('data-letter');
                const phonicSymbol = card.getAttribute('data-phonic');
                const phonicDesc = card.getAttribute('data-phonic-desc');
                const exampleWord = card.querySelector('.example-word').textContent;
                
                const wordSoundButton = card.querySelector('.sound-button:not(.phonic)');
                const phonicSoundButton = card.querySelector('.sound-button.phonic');
                
                // เมื่อคลิกที่การ์ด
                card.addEventListener('click', function() {
                    playLetterSound(letter);
                });
                
                // เมื่อคลิกที่ปุ่มเสียงคำศัพท์
                wordSoundButton.addEventListener('click', function(event) {
                    event.stopPropagation(); // หยุดการ propagate ไปยังการ์ด
                    playWordSound(exampleWord);
                    
                    // เอฟเฟ็กต์ปุ่ม
                    wordSoundButton.classList.add('playing');
                    setTimeout(() => {
                        wordSoundButton.classList.remove('playing');
                    }, 1000);
                });
                
                // เมื่อคลิกที่ปุ่มเสียง Phonics
                phonicSoundButton.addEventListener('click', function(event) {
                    event.stopPropagation(); // หยุดการ propagate ไปยังการ์ด
                    playPhonicSound(phonicSymbol, phonicDesc);
                    
                    // เอฟเฟ็กต์ปุ่ม
                    phonicSoundButton.classList.add('playing');
                    setTimeout(() => {
                        phonicSoundButton.classList.remove('playing');
                    }, 1000);
                });
            });
            
            // สุ่มตัวอักษรเริ่มต้นสำหรับแบบฝึกหัด
            const letters = 'abcdefghijklmnopqrstuvwxyz';
            const randomLetter = letters[Math.floor(Math.random() * letters.length)];
            document.getElementById('target-letter').textContent = randomLetter.toUpperCase();
            
            // ตั้งค่าให้กด Enter ในช่อง input ได้
            const wordInput = document.getElementById('word-input');
            wordInput.addEventListener('keypress', function(event) {
                if (event.key === 'Enter') {
                    checkWord();
                }
            });
        });
