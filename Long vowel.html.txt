            let currentSound = null;
            const soundResult = document.getElementById('sound-result');
            
            // สุ่มคำศัพท์สำหรับเกม
            function getRandomLongVowelWord() {
                const vowels = Object.keys(longVowelSounds);
                const randomVowel = vowels[Math.floor(Math.random() * vowels.length)];
                const words = longVowelSounds[randomVowel];
                const randomWord = words[Math.floor(Math.random() * words.length)];
                
                return {
                    vowel: randomVowel,
                    word: randomWord
                };
            }
            
            // เล่นเสียงสำหรับเกม
            document.getElementById('play-game-sound').addEventListener('click', function() {
                currentSound = getRandomLongVowelWord();
                playSound(currentSound.word, 0.7, 1.0);
                soundResult.textContent = '';
                
                // รีเซ็ตการเลือก
                document.querySelectorAll('.vowel-option').forEach(option => {
                    option.classList.remove('selected');
                });
            });
            
            // เลือกสระ
            document.querySelectorAll('.vowel-option').forEach(option => {
                option.addEventListener('click', function() {
                    // รีเซ็ตการเลือก
                    document.querySelectorAll('.vowel-option').forEach(opt => {
                        opt.classList.remove('selected');
                    });
                    
                    // เลือกอันนี้
                    this.classList.add('selected');
                });
            });
            
            // ตรวจคำตอบ
            document.getElementById('check-answer').addEventListener('click', function() {
                if (!currentSound) {
                    soundResult.textContent = 'กรุณากดฟังเสียงก่อนตรวจสอบ!';
                    soundResult.style.color = '#ff6b6b';
                    return;
                }
                
                const selectedOption = document.querySelector('.vowel-option.selected');
                if (!selectedOption) {
                    soundResult.textContent = 'กรุณาเลือกสระก่อนตรวจสอบ!';
                    soundResult.style.color = '#ff6b6b';
                    return;
                }
                
                const selectedVowel = selectedOption.getAttribute('data-vowel');
                if (selectedVowel === currentSound.vowel) {
                    soundResult.textContent = 'ถูกต้อง! คุณเก่งมาก!';
                    soundResult.style.color = '#4a69bd';
                } else {
                    soundResult.textContent = `ยังไม่ถูกต้อง! คำนี้มีสระเสียงยาว '${currentSound.vowel.toUpperCase()}'`;
                    soundResult.style.color = '#ff6b6b';
                }
            });
            
            // เกมลากและวาง
            const dragResult = document.getElementById('drag-result');
            
            // ตั้งค่า drag and drop
            document.querySelectorAll('.word-option').forEach(option => {
                option.addEventListener('dragstart', function(e) {
                    e.dataTransfer.setData('text/plain', this.getAttribute('data-word'));
                });
            });
            
            document.querySelectorAll('.vowel-group').forEach(group => {
                group.addEventListener('dragover', function(e) {
                    e.preventDefault();
                    this.style.backgroundColor = '#ffeb3b';
                });
                
                group.addEventListener('dragleave', function(e) {
                    this.style.backgroundColor = '';
                });
                
                group.addEventListener('drop', function(e) {
                    e.preventDefault();
                    this.style.backgroundColor = '';
                    const word = e.dataTransfer.getData('text/plain');
                    const correctVowel = document.querySelector(`.word-option[data-word="${word}"]`).getAttribute('data-vowel');
                    const groupVowel = this.getAttribute('data-vowel');
                    
                    if (correctVowel === groupVowel) {
                        // เพิ่มคำลงในกลุ่ม
                        const groupContent = this.querySelector('.group-content');
                        const wordElement = document.createElement('div');
                        wordElement.className = 'word-in-group';
                        wordElement.textContent = word;
                        groupContent.appendChild(wordElement);
                        
                        // ซ่อนคำจากตัวเลือก
                        document.querySelector(`.word-option[data-word="${word}"]`).style.display = 'none';
                        
                        // ตรวจสอบว่าจบเกมหรือยัง
                        if (document.querySelectorAll('.word-option:not([style*="display: none"])').length === 0) {
                            dragResult.textContent = 'ยินดีด้วย! คุณทำได้ถูกต้องทั้งหมด!';
                            dragResult.style.color = '#4a69bd';
                        }
                    } else {
                        dragResult.textContent = 'ลองใหม่อีกครั้ง! คำนี้ไม่ตรงกับกลุ่มสระนี้';
                        dragResult.style.color = '#ff6b6b';
                        setTimeout(() => {
                            dragResult.textContent = '';
                        }, 2000);
                    }
                });
            });
            
            // ฟังก์ชันรีเซ็ตเกม
            function resetDragGame() {
                // ลบคำจากกลุ่มทั้งหมด
                document.querySelectorAll('.word-in-group').forEach(el => el.remove());
                
                // แสดงคำตัวเลือกทั้งหมด
                document.querySelectorAll('.word-option').forEach(option => {
                    option.style.display = 'inline-block';
                });
                
                dragResult.textContent = '';
            }
            
            // ปุ่มตรวจคำตอบ
            document.getElementById('check-drag-game').addEventListener('click', function() {
                // ตรวจสอบว่าคำทั้งหมดถูกวางถูกต้องหรือไม่
                const allWordsPlaced = document.querySelectorAll('.word-option:not([style*="display: none"])').length === 0;
                
                if (allWordsPlaced) {
                    // ตรวจสอบว่าคำถูกวางในกลุ่มที่ถูกต้องทั้งหมด
                    let allCorrect = true;
                    
                    document.querySelectorAll('.vowel-group').forEach(group => {
                        const groupVowel = group.getAttribute('data-vowel');
                        const wordsInGroup = group.querySelectorAll('.word-in-group');
                        
                        wordsInGroup.forEach(wordEl => {
                            const word = wordEl.textContent;
                            const correctVowel = document.querySelector(`.word-option[data-word="${word}"]`).getAttribute('data-vowel');
                            
                            if (correctVowel !== groupVowel) {
                                allCorrect = false;
                            }
                        });
                    });
                    
                    if (allCorrect) {
                        dragResult.textContent = 'ยินดีด้วย! คุณทำได้ถูกต้องทั้งหมด!';
                        dragResult.style.color = '#4a69bd';
                    } else {
                        dragResult.textContent = 'มีบางคำที่อยู่ในกลุ่มที่ไม่ถูกต้อง!';
                        dragResult.style.color = '#ff6b6b';
                    }
                } else {
                    dragResult.textContent = 'กรุณาจับคู่คำศัพท์ให้ครบทุกคำ!';
                    dragResult.style.color = '#ff6b6b';
                }
            });
            
            // ปุ่มรีเซ็ตเกม
            document.getElementById('reset-drag-game').addEventListener('click', resetDragGame);
        });
    </script>
