    <script>
        // ข้อมูลสำหรับเกม
        const blendConsonantWords = [
            { word: "frog", blend: "fr", image: "🐸" },
            { word: "block", blend: "bl", image: "🧱" },
            { word: "clock", blend: "cl", image: "⏰" },
            { word: "flag", blend: "fl", image: "🚩" },
            { word: "grape", blend: "gr", image: "🍇" },
            { word: "prize", blend: "pr", image: "🏆" },
            { word: "bread", blend: "br", image: "🍞" },
            { word: "crab", blend: "cr", image: "🦀" },
            { word: "drum", blend: "dr", image: "🥁" },
            { word: "sky", blend: "sk", image: "🌌" },
            { word: "slide", blend: "sl", image: "🛝" },
            { word: "smile", blend: "sm", image: "😊" },
            { word: "snake", blend: "sn", image: "🐍" },
            { word: "spoon", blend: "sp", image: "🥄" },
            { word: "star", blend: "st", image: "⭐" },
            { word: "swim", blend: "sw", image: "🏊" }
        ];

        const blendConsonants = ["bl", "cl", "fl", "gl", "pl", "br", "cr", "dr", "fr", "gr", "pr", "tr", "sc", "sk", "sl", "sm", "sn", "sp", "st", "sw"];

        // ฟังก์ชันสำหรับเล่นเสียง
        function playSound(text, rate = 0.8, pitch = 1.0) {
            const utterance = new SpeechSynthesisUtterance(text);
            utterance.lang = 'en-US';
            utterance.rate = rate;
            utterance.pitch = pitch;
            window.speechSynthesis.speak(utterance);
        }

        // ฟังก์ชันย้อนกลับ
        function goBack() {
            alert('กำลังกลับไปหน้า Dashboard...');
            // ในเว็บจริง: window.location.href = 'dashboard.html';
        }

        // ตั้งค่า event listeners เมื่อโหลดหน้าเสร็จ
        document.addEventListener('DOMContentLoaded', function() {
            // ตั้งค่า event สำหรับปุ่มเสียง
            document.querySelectorAll('.word-sound').forEach(button => {
                button.addEventListener('click', function() {
                    const word = this.getAttribute('data-word');
                    playSound(word, 0.7, 1.0);
                    
                    // เอฟเฟ็กต์ปุ่ม
                    this.style.backgroundColor = '#ff6b6b';
                    setTimeout(() => {
                        this.style.backgroundColor = '';
                    }, 500);
                });
            });
            
            // เกมเติมพยัญชนะผสม
            let currentPuzzle = null;
            const puzzleImage = document.getElementById('puzzle-image');
            const consonantBlank = document.getElementById('consonant-blank');
            const puzzleResult = document.getElementById('puzzle-result');
            let selectedBlend = null;
            
            // สุ่มปริศนา
            function generatePuzzle() {
                const randomIndex = Math.floor(Math.random() * blendConsonantWords.length);
                currentPuzzle = blendConsonantWords[randomIndex];
                
                // แสดงภาพ
                puzzleImage.textContent = currentPuzzle.image;
                
                // รีเซ็ตการเลือก
                puzzleResult.textContent = '';
                selectedBlend = null;
                document.querySelectorAll('.blend-option').forEach(option => {
                    option.classList.remove('selected');
                });
            }
            
            // เลือกพยัญชนะผสม
            document.querySelectorAll('.blend-option').forEach(option => {
                option.addEventListener('click', function() {
                    // รีเซ็ตการเลือก
                    document.querySelectorAll('.blend-option').forEach(opt => {
                        opt.classList.remove('selected');
                    });
                    
                    // เลือกอันนี้
                    this.classList.add('selected');
                    selectedBlend = this.getAttribute('data-blend');
                    consonantBlank.textContent = selectedBlend;
                });
            });
            
            // ตรวจสอบคำตอบ
            document.getElementById('check-puzzle').addEventListener('click', function() {
                if (!selectedBlend) {
                    puzzleResult.textContent = 'กรุณาเลือกพยัญชนะผสมก่อนตรวจสอบ!';
                    puzzleResult.style.color = '#ff6b6b';
                    return;
                }
                
                if (selectedBlend === currentPuzzle.blend) {
                    puzzleResult.textContent = 'ถูกต้อง! เยี่ยมมาก!';
                    puzzleResult.style.color = '#4a69bd';
                    playSound(currentPuzzle.word, 0.7, 1.0);
                } else {
                    puzzleResult.textContent = 'ยังไม่ถูกต้อง! ลองใหม่อีกครั้ง';
                    puzzleResult.style.color = '#ff6b6b';
                }
            });
            
            // คำถัดไป
            document.getElementById('next-puzzle').addEventListener('click', generatePuzzle);
            
            // เริ่มเกมแรก
            generatePuzzle();
            
            // เกมสร้างคำจากพยัญชนะผสม
            const targetBlend = document.getElementById('target-blend');
            const currentBlend = document.getElementById('current-blend');
            const wordInput = document.getElementById('word-input');
            const wordResult = document.getElementById('word-result');
            let currentBlendType = "bl";
            
            // สุ่มพยัญชนะผสมใหม่
            function generateNewBlend() {
                const randomIndex = Math.floor(Math.random() * blendConsonants.length);
                currentBlendType = blendConsonants[randomIndex];
                targetBlend.textContent = currentBlendType;
                currentBlend.textContent = currentBlendType;
                wordInput.value = '';
                wordResult.textContent = '';
            }
            
            // ตรวจสอบคำที่ผู้ใช้ป้อน
            document.getElementById('check-word').addEventListener('click', function() {
                const word = wordInput.value.trim().toLowerCase();
                
                if (!word) {
                    wordResult.textContent = 'กรุณาพิมพ์คำศัพท์ก่อนตรวจสอบ!';
                    wordResult.style.color = '#ff6b6b';
                    return;
                }
                
                if (word.startsWith(currentBlendType)) {
                    // ตรวจสอบว่าคำนี้มีอยู่จริงหรือไม่
                    const isValidWord = blendConsonantWords.some(item => item.word === word) || 
                                      word.length >= 3; // อนุญาตให้ใช้คำอื่นๆ ที่ขึ้นต้นด้วย blend นั้นๆ
                    
                    if (isValidWord) {
                        wordResult.textContent = 'ถูกต้อง! คุณเก่งมาก!';
                        wordResult.style.color = '#4a69bd';
                        playSound(word, 0.7, 1.0);
                    } else {
                        wordResult.textContent = 'เป็นคำที่ดี! แต่ลองหาคำอื่นดูนะ';
                        wordResult.style.color = '#4a69bd';
                    }
                } else {
                    wordResult.textContent = 'คำนี้ไม่ขึ้นต้นด้วย ' + currentBlendType + '! ลองใหม่อีกครั้ง';
                    wordResult.style.color = '#ff6b6b';
                }
            });
            
            // ปุ่มสุ่มพยัญชนะผสมใหม่
            document.getElementById('new-blend').addEventListener('click', generateNewBlend);
            
            // เริ่มเกมแรก
            generateNewBlend();
        });
    </script>
